# frpc

```agsl
wget https://github.com/fatedier/frp/releases/download/v0.61.2/frp_0.61.2_linux_amd64.tar.gz
解压

命令启动
./frpc -c ./frpc.toml


# systemd 启动  frps.service和frpc.service 差不多
yum install systemd

$ sudo vim /etc/systemd/system/frps.service
[Unit]
# 服务名称，可自定义
Description = frp server
After = network.target syslog.target
Wants = network.target

[Service]
Type = simple
# 启动frps的命令，需修改为您的frps的安装路径
ExecStart = /path/to/frps -c /path/to/frps.toml

[Install]
WantedBy = multi-user.target


#启动
systemctl daemon-reload


# 启动frp
sudo systemctl start frps
# 停止frp
sudo systemctl stop frps
# 重启frp
sudo systemctl restart frps
# 查看frp状态
sudo systemctl status frps
#  开机自启
sudo systemctl enable frps

```

启动脚本 win

``` bat

@echo off
setlocal enabledelayedexpansion

:: ===== 配置区域 =====
set FRP_DIR=F:\frp
set FRPC_EXE=frpc.exe
set FRPC_CONF=frpc.toml
set LOG_DIR=%FRP_DIR%\logs
set LOG_FILE=%LOG_DIR%\frpc.log
set PROC_NAME=frpc.exe
:: ====================

if not exist "%LOG_DIR%" (
    mkdir "%LOG_DIR%"
)

cd /d "%FRP_DIR%"

if "%1"=="" goto help
if "%1"=="start" goto start
if "%1"=="stop" goto stop
if "%1"=="status" goto status
if "%1"=="log" goto log
goto help

:start
tasklist | findstr /i "%PROC_NAME%" >nul
if %errorlevel%==0 (
    echo [INFO] frpc is already running.
    goto eof
)

echo [INFO] Starting frpc...
start "frpc" /min "%FRP_DIR%\%FRPC_EXE%" -c "%FRP_DIR%\%FRPC_CONF%" >> "%LOG_FILE%" 2>&1
timeout /t 2 >nul

tasklist | findstr /i "%PROC_NAME%" >nul
if %errorlevel%==0 (
    echo [OK] frpc started successfully.
) else (
    echo [ERROR] frpc failed to start.
)
goto eof

:stop
tasklist | findstr /i "%PROC_NAME%" >nul
if not %errorlevel%==0 (
    echo [INFO] frpc is not running.
    goto eof
)

echo [INFO] Stopping frpc...
taskkill /f /im "%PROC_NAME%" >nul 2>&1
echo [OK] frpc stopped.
goto eof

:status
tasklist | findstr /i "%PROC_NAME%" >nul
if %errorlevel%==0 (
    echo [RUNNING] frpc is running.
) else (
    echo [STOPPED] frpc is not running.
)
goto eof

:log
if not exist "%LOG_FILE%" (
    echo [INFO] Log file not found.
    goto eof
)
type "%LOG_FILE%"
goto eof

:help
echo Usage:
echo   frpc.bat start    - Start frpc
echo   frpc.bat stop     - Stop frpc
echo   frpc.bat status   - Check status
echo   frpc.bat log      - View log
goto eof

:eof
endlocal

```
